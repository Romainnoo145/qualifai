**Step 1: Extend NarrativeSection in types.ts**

Add three optional fields to `NarrativeSection`:

```typescript
export type VisualType = 'quote' | 'comparison' | 'signals' | 'stats';

export type NarrativeSection = {
  id: string;
  title: string;
  body: string;
  citations: string[];
  punchline?: string; // One-sentence headline for the brochure page
  visualType?: VisualType; // AI chooses based on evidence found
  visualData?: VisualData; // Structured data matching the visualType
};

export type VisualData =
  | { type: 'quote'; quote: string; attribution: string }
  | {
      type: 'comparison';
      items: Array<{ label: string; before: string; after: string }>;
    }
  | {
      type: 'signals';
      items: Array<{
        label: string;
        value: string;
        trend: 'up' | 'down' | 'neutral';
      }>;
    }
  | {
      type: 'stats';
      items: Array<{ label: string; value: string; context?: string }>;
    };
```

Fields are optional so existing analysis-v2 data (without visual fields) still parses without error. New analyses will include them.

// Import VisualData type
import type { NarrativeSection, VisualData } from '@/lib/analysis/types';

// Visual renderer component (add inside the file)
function SectionVisual({ section }: { section: NarrativeSection }) {
  if (!section.visualData) return null;
  const data = section.visualData;

  if (data.type === 'quote') {
    return (
      <blockquote
        style={{
          borderLeft: `3px solid ${GOLD_MID}`,
          paddingLeft: '24px',
          margin: '40px 0',
          maxWidth: '700px',
        }}
      >
        <p
          style={{
            fontSize: 'clamp(20px, 2.5vw, 28px)',
            fontWeight: 300,
            lineHeight: 1.5,
            fontStyle: 'italic',
            color: TEXT_ON_NAVY,
          }}
        >
          &ldquo;{data.quote}&rdquo;
        </p>
        <cite
          style={{
            display: 'block',
            marginTop: '12px',
            fontSize: '12px',
            fontWeight: 500,
            color: TEXT_MUTED_ON_NAVY,
            fontStyle: 'normal',
          }}
        >
          — {data.attribution}
        </cite>
      </blockquote>
    );
  }

  if (data.type === 'stats') {
    return (
      <div
        style={{
          display: 'grid',
          gridTemplateColumns: `repeat(${Math.min(data.items.length, 3)}, 1fr)`,
          gap: '32px',
          margin: '40px 0',
        }}
      >
        {data.items.map((item, i) => (
          <div key={i} style={{ textAlign: 'left' }}>
            <div
              style={{
                fontSize: 'clamp(32px, 4vw, 48px)',
                fontWeight: 700,
                ...goldGradientText,
                lineHeight: 1,
                marginBottom: '8px',
              }}
            >
              {item.value}
            </div>
            <div
              style={{
                fontSize: '13px',
                fontWeight: 500,
                color: TEXT_ON_NAVY,
                marginBottom: '4px',
              }}
            >
              {item.label}
            </div>
            {item.context && (
              <div
                style={{
                  fontSize: '11px',
                  color: TEXT_MUTED_ON_NAVY,
                }}
              >
                {item.context}
              </div>
            )}
          </div>
        ))}
      </div>
    );
  }

  if (data.type === 'comparison') {
    return (
      <div style={{ margin: '40px 0' }}>
        {data.items.map((item, i) => (
          <div
            key={i}
            style={{
              display: 'grid',
              gridTemplateColumns: '1fr auto 1fr',
              gap: '16px',
              alignItems: 'center',
              padding: '16px 0',
              borderBottom:
                i < data.items.length - 1
                  ? `1px solid ${CONTAINER_BORDER}`
                  : 'none',
            }}
          >
            <div style={{ fontSize: '14px', color: TEXT_MUTED_ON_NAVY }}>
              {item.before}
            </div>
            <div
              style={{
                fontSize: '11px',
                fontWeight: 500,
                letterSpacing: '0.14em',
                textTransform: 'uppercase',
                color: GOLD_MID,
              }}
            >
              {item.label}
            </div>
            <div
              style={{
                fontSize: '14px',
                fontWeight: 500,
                color: TEXT_ON_NAVY,
                textAlign: 'right',
              }}
            >
              {item.after}
            </div>
          </div>
        ))}
      </div>
    );
  }

  if (data.type === 'signals') {
    return (
      <div
        style={{
          margin: '40px 0',
          display: 'flex',
          flexDirection: 'column',
          gap: '20px',
        }}
      >
        {data.items.map((item, i) => (
          <div
            key={i}
            style={{
              display: 'flex',
              alignItems: 'center',
              gap: '16px',
            }}
          >
            <span
              style={{
                fontSize: '20px',
                width: '32px',
                textAlign: 'center',
              }}
            >
              {item.trend === 'up'
                ? '\u2191'
                : item.trend === 'down'
                  ? '\u2193'
                  : '\u2192'}
            </span>
            <div>
              <div
                style={{
                  fontSize: 'clamp(18px, 2vw, 24px)',
                  fontWeight: 700,
                  color: TEXT_ON_NAVY,
                }}
              >
                {item.value}

                }}
              >
                {item.label}
              </div>
            </div>
          </div>
        ))}
      </div>
    );
  }

  return null;
}
