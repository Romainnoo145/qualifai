```typescript
/**
 * IMPORT-01..04 — One-shot migration script for klarifai-core YAML data.
 *
 * Imports:
 *   1. klarifai-core/data/clients/*.yaml -> Qualifai Prospect (idempotent on readableSlug)
 *   2. klarifai-core/data/quotes/{year}/*.yaml -> Qualifai Quote + QuoteLine (idempotent on nummer)
 *
 * Usage:
 *   tsx scripts/import-klarifai-yaml.ts                 # DRY (default) — logs intended writes
 *   tsx scripts/import-klarifai-yaml.ts --apply         # writes to DB
 *   tsx scripts/import-klarifai-yaml.ts --source /path  # override default source dir
 *
 * The script MUST be run with the singleton Prisma client (lib/prisma.ts).
 * Do NOT instantiate `new PrismaClient()` — Prisma 7 adapter quirk.
 *
 * Contact import is deferred to Phase 61 admin UI (research recommendation #6).
 */
import 'dotenv/config';
import { readFileSync, readdirSync, existsSync } from 'node:fs';
import { join, resolve } from 'node:path';
import { parse as parseYaml } from 'yaml';
import { z } from 'zod';
import { prisma } from '@/lib/prisma';

// ---------------------------------------------------------------------------
// CLI argument parsing
// ---------------------------------------------------------------------------

const argv = process.argv.slice(2);
const APPLY = argv.includes('--apply');
const sourceFlagIdx = argv.indexOf('--source');
const DEFAULT_SOURCE = resolve(__dirname, '../../klarifai-core/data');
const SOURCE_DIR =
  sourceFlagIdx > -1 && argv[sourceFlagIdx + 1]
    ? argv[sourceFlagIdx + 1]
    : DEFAULT_SOURCE;
const PROJECT_SLUG = process.env.IMPORT_PROJECT_SLUG ?? 'klarifai';

// ---------------------------------------------------------------------------
// YAML schemas — match real fixture shapes
// ---------------------------------------------------------------------------

const ClientYamlSchema = z.object({
  naam: z.string(),
  slug: z.string(), // human slug like "marfa" — maps to readableSlug
  contactpersoon: z.string().optional(),
  email: z.string().email().optional(),
  adres: z
    .object({
      straat: z.string().optional(),
      postcode: z.string().optional(),
      stad: z.string().optional(),
    })
    .optional(),
  standaard_tarief: z.number().optional(),
  betaaltermijn_dagen: z.number().optional(),
});
type ClientYaml = z.infer<typeof ClientYamlSchema>;

const QuoteLineYamlSchema = z.object({
  fase: z.string(),
  omschrijving: z.string(),
  oplevering: z.string(),
  uren: z.number().int().nonnegative(),
  tarief: z.number().int(), // SIGNED — negative allowed for OFF003 Pakketkorting
});

const QuoteYamlSchema = z.object({
  nummer: z.string(),
  datum: z.string(), // ISO date
  geldig_tot: z.string(),
  klant: z.string(), // matches ClientYaml.slug -> Prospect.readableSlug
  status: z.string().default('concept'),
  onderwerp: z.string(),
  tagline: z.string().optional().default(''),
  introductie: z.string().optional().default(''),
  uitdaging: z.string().optional().default(''),
  aanpak: z.string().optional().default(''),
  regels: z.array(QuoteLineYamlSchema),
  btw_percentage: z.number(),
  scope: z.string().optional().default(''),
  buiten_scope: z.string().optional().default(''),
});
type QuoteYaml = z.infer<typeof QuoteYamlSchema>;

// ---------------------------------------------------------------------------
// Helpers
// ---------------------------------------------------------------------------

function log(
  action: 'create' | 'update' | 'skip' | 'warn' | 'info',
  kind: string,
  label: string,
  detail = '',
) {
  const tag = APPLY ? '[APPLY]' : '[DRY]  ';
  console.log(
    `${tag} ${action.toUpperCase().padEnd(6)} ${kind.padEnd(10)} ${label} ${detail}`,
  );
}

function computeNetto(regels: QuoteYaml['regels']): number {
  return regels.reduce((sum, r) => sum + r.uren * r.tarief, 0);
}
function computeBruto(netto: number, btwPercentage: number): number {
  return Math.round(netto * (1 + btwPercentage / 100) * 100) / 100;
}

// ---------------------------------------------------------------------------
// Main
// ---------------------------------------------------------------------------

async function main() {
  console.log('---');
  console.log(
    `[import-klarifai-yaml] mode=${APPLY ? 'APPLY' : 'DRY'} source=${SOURCE_DIR}`,
  );
  console.log(`[import-klarifai-yaml] project=${PROJECT_SLUG}`);
  console.log('---');

  if (!existsSync(SOURCE_DIR)) {
    throw new Error(
      `Source directory does not exist: ${SOURCE_DIR}. Pass --source <path> to override.`,
    );
  }

  // Resolve the active Project (multi-tenant scope target)
  const project = await prisma.project.findUnique({
    where: { slug: PROJECT_SLUG },
  });
  if (!project) {
    throw new Error(
      `Project not found in DB: slug=${PROJECT_SLUG}. Seed it first.`,
    );
  }
  log('info', 'project', project.slug, `id=${project.id}`);

  // -------------------------------------------------------------------------
  // 1. Clients -> Prospects
  // -------------------------------------------------------------------------
  const clientsDir = join(SOURCE_DIR, 'clients');
  if (!existsSync(clientsDir)) {
    log('warn', 'clients', '(dir missing)', clientsDir);
  } else {
    const clientFiles = readdirSync(clientsDir).filter(
      (f) => f.endsWith('.yaml') || f.endsWith('.yml'),
    );
    for (const file of clientFiles) {
      const raw = parseYaml(readFileSync(join(clientsDir, file), 'utf8'));
      const parsed = ClientYamlSchema.safeParse(raw);
      if (!parsed.success) {
        log(
          'warn',
          'client',
          file,
          `parse error: ${parsed.error.issues.map((i) => i.path.join('.')).join(',')}`,
        );
        continue;
      }
      await upsertClient(parsed.data, project.id, file);

      log(
        'warn',
        'contact',
        file,
        'Contact import deferred to Phase 61 admin UI (research rec #6)',
      );
    }
  }

  // -------------------------------------------------------------------------
  // 2. Quotes -> Quote + QuoteLine
  // -------------------------------------------------------------------------
  const quotesRoot = join(SOURCE_DIR, 'quotes');
  if (!existsSync(quotesRoot)) {
    log('warn', 'quotes', '(dir missing)', quotesRoot);
  } else {
    const yearDirs = readdirSync(quotesRoot).filter((d) => /^\d{4}$/.test(d));
    for (const year of yearDirs) {
      const yearDir = join(quotesRoot, year);
      const quoteFiles = readdirSync(yearDir).filter(
        (f) => f.endsWith('.yaml') || f.endsWith('.yml'),
      );
      for (const file of quoteFiles) {
        const raw = parseYaml(readFileSync(join(yearDir, file), 'utf8'));
        const parsed = QuoteYamlSchema.safeParse(raw);
        if (!parsed.success) {
          log(
            'warn',
            'quote',
            file,
            `parse error: ${parsed.error.issues.map((i) => i.path.join('.')).join(',')}`,
          );
          continue;
        }
        await upsertQuote(parsed.data, project.id, file);
      }
    }
  }

  console.log('---');
  if (!APPLY) {
    console.log(
      '[import-klarifai-yaml] DRY RUN COMPLETE — no writes performed. Use --apply to commit.',
    );
  } else {
    console.log('[import-klarifai-yaml] APPLY COMPLETE.');
  }
  console.log('---');
}

async function upsertClient(
  client: ClientYaml,
  projectId: string,
  file: string,
) {
  // Idempotency: match on readableSlug (research rec #2 — NOT the 12-char CUID slug)
  const existing = await prisma.prospect.findUnique({
    where: { readableSlug: client.slug },
  });

  if (existing) {
    log(
      'update',
      'prospect',
      client.slug,
      `id=${existing.id} (matched by readableSlug)`,
    );
    if (APPLY) {
      await prisma.prospect.update({
        where: { id: existing.id },
        data: {
          companyName: client.naam,
          city: client.adres?.stad ?? existing.city,
        },
      });
    }
    return existing;
  }

  // Create new — domain is NOT NULL on Prospect, use a placeholder if YAML has none
  const placeholderDomain = `${client.slug}.placeholder.invalid`;
  log('create', 'prospect', client.slug, `domain=${placeholderDomain}`);
  if (APPLY) {
    return prisma.prospect.create({
      data: {
        readableSlug: client.slug,
        companyName: client.naam,
        domain: placeholderDomain,
        city: client.adres?.stad,
        projectId,
        // status defaults to DRAFT
      },
    });
  }
  return null;
}

async function upsertQuote(quote: QuoteYaml, projectId: string, file: string) {
  const netto = computeNetto(quote.regels);
  const bruto = computeBruto(netto, quote.btw_percentage);
  const totalLabel = `netto=${netto} bruto=${bruto.toFixed(2)}`;

  // Idempotency: match on global nummer (research rec #3)
  const existing = await prisma.quote.findUnique({
    where: { nummer: quote.nummer },
  });

  // Find the Prospect by readableSlug = klant
  const prospect = await prisma.prospect.findUnique({
    where: { readableSlug: quote.klant },
  });
  if (!prospect) {
    log(
      'warn',
      'quote',
      quote.nummer,
      `no prospect found for klant=${quote.klant} — skipping`,
    );
    return;
  }

  if (existing) {
    log(
      'skip',
      'quote',
      quote.nummer,
      `${totalLabel} (already in DB, idempotent skip)`,
    );
    return existing;
  }

  log('create', 'quote', quote.nummer, totalLabel);
  if (APPLY) {
    return prisma.quote.create({
      data: {
        nummer: quote.nummer,
        datum: new Date(quote.datum),
        geldigTot: new Date(quote.geldig_tot),
        onderwerp: quote.onderwerp,
        tagline: quote.tagline,
        introductie: quote.introductie,
        uitdaging: quote.uitdaging,
        aanpak: quote.aanpak,
        btwPercentage: quote.btw_percentage,
        scope: quote.scope,
        buitenScope: quote.buiten_scope,
        prospect: { connect: { id: prospect.id } },
        // status defaults to DRAFT
        lines: {
          create: quote.regels.map((r, idx) => ({
            fase: r.fase,
            omschrijving: r.omschrijving,
            oplevering: r.oplevering,
            uren: r.uren,
            tarief: r.tarief,
            position: idx,
          })),
        },
      },
      include: { lines: true },
    });
  }
  return null;
}

main()
  .then(() => process.exit(0))
  .catch((err) => {
    console.error('[import-klarifai-yaml] FAILED');
    console.error(err);
    process.exit(1);
  });
```
