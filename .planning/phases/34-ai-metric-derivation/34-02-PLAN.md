const clampInt = (
  val: unknown,
  lo: number,
  hi: number,
  def: number,
): number => {
  if (typeof val !== 'number' || isNaN(val)) return def;
  return Math.round(Math.min(hi, Math.max(lo, val)));
};
const clampFloat = (
  val: unknown,
  lo: number,
  hi: number,
  def: number,
): number => {
  if (typeof val !== 'number' || isNaN(val)) return def;
  return Math.min(hi, Math.max(lo, val));
};
const VALID_SOURCE_TYPES = new Set([

]);
