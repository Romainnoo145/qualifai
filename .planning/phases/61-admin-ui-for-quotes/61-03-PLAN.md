    import { cn } from '@/lib/utils';

    interface Props {
      createdAt: Date | string;
      snapshotAt: Date | string | null;
      viewedAt?: Date | string | null;
      acceptedAt?: Date | string | null;
    }

    const FORMATTER = new Intl.DateTimeFormat('nl-NL', {
      day: 'numeric',
      month: 'long',
      year: 'numeric',
      hour: '2-digit',
      minute: '2-digit',
    });

    function formatOrPlaceholder(value: Date | string | null | undefined): string {
      if (!value) return 'â€” nog niet';
      return FORMATTER.format(new Date(value));
    }

    export function QuoteStatusTimeline({ createdAt, snapshotAt, viewedAt = null, acceptedAt = null }: Props) {
      const slots: { label: string; value: Date | string | null | undefined }[] = [
        { label: 'Aangemaakt', value: createdAt },
        { label: 'Verstuurd', value: snapshotAt },
        { label: 'Bekeken', value: viewedAt },
        { label: 'Geaccepteerd', value: acceptedAt },
      ];

      return (
        <div className="glass-card p-6">
          <h3 className="mb-4 text-sm font-black uppercase tracking-wide text-slate-500">
            Tijdlijn
          </h3>
          <ol className="flex flex-col gap-4 md:flex-row md:items-start md:gap-6">
            {slots.map((slot, idx) => {
              const filled = Boolean(slot.value);
              return (
                <li key={slot.label} className="flex flex-col gap-1">
                  <span
                    className={cn(
                      'text-xs font-bold uppercase tracking-wide',
                      filled ? 'text-[#040026]' : 'text-slate-400',
                    )}
                  >
                    {slot.label}
                  </span>
                  <span
                    className={cn(
                      'text-sm',
                      filled ? 'font-black text-[#040026]' : 'font-semibold text-slate-400',
                    )}
                  >
                    {formatOrPlaceholder(slot.value)}
                  </span>
                </li>
              );
            })}
          </ol>
          <p className="mt-4 text-xs text-slate-400">
            Bekeken en Geaccepteerd worden vanaf Phase 62 automatisch bijgewerkt.
          </p>
        </div>
      );

    import prisma from '@/lib/prisma';
    import { resolveAdminProjectScope } from '@/server/admin-auth';

    // Next.js App Router route handler â€” admin-only preview.
    // Scope check: token â†’ allowedProjectSlug â†’ project.id â†’ prospect.projectId filter.
    // NEVER fetches a Quote without the prospect.projectId scope (multi-tenant boundary per Phase 60-04).

    export async function GET(
      req: NextRequest,
      context: { params: { id: string } | Promise<{ id: string }> },
    ) {
      // Next 15 ships both sync and Promise params depending on compat mode â€” handle both.
      const rawParams = await Promise.resolve(context.params);
      const id = rawParams.id;

      const token =
        req.headers.get('x-admin-token') ??
        req.nextUrl.searchParams.get('token');
      const scope = resolveAdminProjectScope(token);
      if (!scope) {
        return new Response('Unauthorized', { status: 401 });
      }

      const project = await prisma.project.findUnique({
        where: { slug: scope.allowedProjectSlug },
        select: { id: true },
      });
      if (!project) {
        return new Response(`Unknown project scope: ${scope.allowedProjectSlug}`, {
          status: 400,
        });
      }

      const quote = await prisma.quote.findFirst({
        where: {
          id,
          prospect: { projectId: project.id },
        },
        include: {
          lines: { orderBy: { position: 'asc' } },
          prospect: { select: { slug: true, companyName: true } },
        },
      });
      if (!quote) {
        return new Response('Not Found', { status: 404 });
      }

      const html = await renderQuotePreview(quote);

      return new Response(html, {
        headers: {
          'content-type': 'text/html; charset=utf-8',
          'cache-control': 'no-store',
          'x-robots-tag': 'noindex, nofollow',
          'referrer-policy': 'no-referrer',
        },
      });
    }
